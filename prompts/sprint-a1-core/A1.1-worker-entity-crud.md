  id: StaffId

  email?: Email
  phone?: Phone

  updatedAt: Date

export class TempWorker {
  private constructor(private readonly props: TempWorkerProps) {}

  static create(input: CreateInput, clock: Clock): TempWorker {

## Tests à écrire

### Unit (domaine + app)

- **Avs** : checksum valide, invalide (mauvais chiffre), format incorrect (longueur), valeurs limites.
- **Iban** : mod 97 valide, invalide, CH seul accepté, longueur.
- **TempWorker.create** : chemin heureux, rename, archive idempotent.
- **RegisterWorkerUseCase** :
  - ✓ chemin heureux (worker créé + audit log écrit)
  - ✗ AVS invalide → InvalidAvs
  - ✗ IBAN invalide → InvalidIban
  - ✗ AVS déjà existant pour cette agence → DuplicateAvs
  - ✓ même AVS dans agence B → OK (isolation tenant)
- **UpdateWorkerUseCase** : champs modifiables uniquement, refus de modifier AVS.
- **ArchiveWorkerUseCase** : archivage = soft delete, relecture GET → 404, relecture via paramètre `includeArchived=true` → OK.

### Intégration (Testcontainers Postgres)

- Insertion via repository, lecture, update.
- Contrainte unique AVS par (agency_id, avs) fonctionne.
- Index sur (agency_id, deleted_at) utilisé (EXPLAIN).

### API (Supertest)

- POST /workers avec valid input → 201 + Location.
- POST sans auth → 401.
- POST avec rôle `viewer` → 403.
- POST avec Idempotency-Key → 2e appel retourne la réponse cached.
- GET /workers/:id en cross-tenant → 404 (pas 403, pour ne pas leak l'existence).
- PUT /workers/:id avec id inexistant → 404.
